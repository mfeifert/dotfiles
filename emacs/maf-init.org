#+begin_src emacs-lisp :tangle "~/.config/emacs/maf-init.el"
(defun maf-ledger ()
  "Open ledger file and ledger-report windows."
  (interactive)
  (delete-other-windows)
  (find-file "~/sync/finance/current_year.ledger")
  (ledger-report "Balance Sheet" nil)
  (other-window 1)
  (message nil))
